Wonderfruit F&B Task Manager
Connect to Google Sheets so Korn, Michel and Matthieu share live data. One-time setup, ~3 minutes.
1
Create a Google Sheet
Go to sheets.google.com, create a new blank spreadsheet, and name it WF F&B Tasks.
2
Open Apps Script
In the sheet, click Extensions → Apps Script. Delete all existing code in the editor.
3
Paste this backend code
Copy the code below, paste it into the Apps Script editor, and press Ctrl+S to save.
const SHEET_NAME = 'WF_FNB_DATA';
function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let s = ss.getSheetByName(SHEET_NAME);
  if (!s) { s = ss.insertSheet(SHEET_NAME); s.getRange('A1').setValue('{}'); }
  return s;
}
function doGet(e) {
  const raw = getSheet().getRange('A1').getValue();
  const data = raw ? JSON.parse(raw) : { projects: [], uid: 1 };
  return ContentService.createTextOutput(JSON.stringify({ ok: true, data }))
    .setMimeType(ContentService.MimeType.JSON);
}
function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    if (body.action === 'write') {
      getSheet().getRange('A1').setValue(JSON.stringify(body.data));
      return ContentService.createTextOutput(JSON.stringify({ ok: true }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ ok:false, error:err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
4
Deploy as a web app
Click Deploy → New deployment. Set type to Web app, execute as Me, access to Anyone. Click Deploy, authorize, then copy the Web app URL.
5
Paste the URL here
All three team members do this step once using the same URL.
